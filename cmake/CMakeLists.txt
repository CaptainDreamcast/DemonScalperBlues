cmake_minimum_required(VERSION 3.25) 
project(DemonScalperBlues CXX)               
set(CMAKE_CXX_STANDARD 17)            
set(CMAKE_MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>")

include("${CMAKE_SOURCE_DIR}/../../addons/prism/cmake/prism.cmake" REQUIRED)

# Add source/header files to project
file(GLOB_RECURSE SOURCES ../*.cpp SOURCES ../*.h SOURCES ../*.rc)
list(FILTER SOURCES EXCLUDE REGEX ".*web.*")
list(FILTER SOURCES EXCLUDE REGEX ".*/build/.*")

add_link_options(/NODEFAULTLIB:libcmt.lib)
add_link_options(/IGNORE:4099,4286,4098)
add_link_options(/OPT:REF)
add_link_options(/OPT:ICF)

# Define exe
add_executable(DemonScalperBlues ${SOURCES})

# Link with libraries
target_link_libraries(DemonScalperBlues prism)
target_link_libraries(DemonScalperBlues SDL2)
target_link_libraries(DemonScalperBlues SDL2main)
target_link_libraries(DemonScalperBlues SDL2_image)
target_link_libraries(DemonScalperBlues SDL2_mixer)
target_link_libraries(DemonScalperBlues SDL2_ttf)
target_link_libraries(DemonScalperBlues enet)
target_link_libraries(DemonScalperBlues freetype)
target_link_libraries(DemonScalperBlues glew)
target_link_libraries(DemonScalperBlues curl)
target_link_libraries(DemonScalperBlues libjpeg)
target_link_libraries(DemonScalperBlues libpng)
target_link_libraries(DemonScalperBlues libwebp)
target_link_libraries(DemonScalperBlues zstd)
target_link_libraries(DemonScalperBlues zlib)
target_link_libraries(DemonScalperBlues crypt32)
target_link_libraries(DemonScalperBlues wldap32)
target_link_libraries(DemonScalperBlues ws2_32)
target_link_libraries(DemonScalperBlues winmm)
target_link_libraries(DemonScalperBlues OpenGL32)
target_link_libraries(DemonScalperBlues version)

# Define preprocessor definitions
target_compile_definitions(DemonScalperBlues PUBLIC UNICODE)
target_compile_definitions(DemonScalperBlues PUBLIC _UNICODE)

set_property(TARGET DemonScalperBlues PROPERTY INTERPROCEDURAL_OPTIMIZATION TRUE)
target_compile_options(DemonScalperBlues PRIVATE /Gy)

set_target_properties(DemonScalperBlues PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${CMAKE_SOURCE_DIR}/../assets)
set_target_properties(DemonScalperBlues PROPERTIES RUNTIME_OUTPUT_DIRECTORY_DEBUG ${CMAKE_SOURCE_DIR}/../assets)

# Copy over DLLs
file(GLOB DLLS "${CMAKE_SOURCE_DIR}/../../addons/prism/windows/vs17/DLL/*.dll")
foreach(DLL ${DLLS})
  add_custom_command(TARGET DemonScalperBlues POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
    ${DLL}
    $<TARGET_FILE_DIR:DemonScalperBlues>)
endforeach()

# Optimize for release build
if(CMAKE_BUILD_TYPE STREQUAL "Release")
  target_compile_options(DemonScalperBlues PRIVATE /O2)
endif()
